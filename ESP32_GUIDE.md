# Panduan Integrasi ESP32 ke Next.js API & Supabase Database (Multi-Tabel)

Dokumen ini berisi penjelasan cara menghubungkan sensor ESP32 ke API endpoint Next.js yang telah terproteksi token dan terhubung ke **dua tabel** database Supabase (`public.sensor` dan `public.system_logs`).

---

## 1. Skema Tabel Database

### A. Tabel `public.sensor` (Untuk Data Kelembapan & Cuaca)
```sql
create table public.sensor (
  id bigint generated by default as identity not null,
  moisture character varying not null,
  humidity character varying null,
  temperature character varying null,
  constraint sensor_pkey primary key (id)
) TABLESPACE pg_default;
```

### B. Tabel `public.system_logs` (Untuk Status/Kesehatan Hardware)
```sql
create table public.system_logs (
  id bigint generated by default as identity not null,
  device_id character varying not null,
  cpu_temperature character varying null,
  hall_magnetic character varying null,
  touch_raw character varying null,
  constraint system_logs_pkey primary key (id)
) TABLESPACE pg_default;
```

## 4. Contoh Kode Arduino/C++ untuk ESP32 (Multi-Tabel)
Berikut adalah kode program lengkap untuk ESP32. Kode ini secara otomatis membaca data analog dari sensor kelembapan tanah, serta suhu CPU internal, sensor Hall magnetik internal, dan Pin sentuh (Touch Pin).

```cpp
#include <WiFi.h>
#include <HTTPClient.h>
#include <ArduinoJson.h> // Membutuhkan library ArduinoJson (v6 atau v7)

// ----------------------------------------------------
// KONFIGURASI WIFI & API
// ----------------------------------------------------
const char* ssid = "SSID_WIFI_KAMU";           // Ganti dengan nama WiFi kamu
const char* password = "PASSWORD_WIFI_KAMU";   // Ganti dengan password WiFi kamu

// Gunakan alamat IP laptop/PC kamu jika menjalankan Next.js secara lokal.
const char* serverUrl = "http://192.168.1.8:3000/api/sensor"; // Ganti IP sesuai IP laptop kamu (contoh: 192.168.1.8)
const char* apiToken = "bdfe0835301faa88f2f983y3498yt90b5cf8af0f2d163c6f67877c684c03fe90c1"; // Token dari file .env

// Pin sensor kelembapan tanah (Analog Pin)
const int soilPin = 34; 

// Touch Pin T0 (GPIO 4 pada ESP32) digunakan untuk mengukur sensor sentuh
const int touchPin = 4; 

#ifdef __cplusplus
extern "C" {
#endif
uint8_t temprature_sens_read(); // deklarasi fungsi baca suhu internal ESP32
#ifdef __cplusplus
}
#endif

void setup() {
  Serial.begin(115200);
  delay(1000);

  // Konek ke WiFi
  WiFi.begin(ssid, password);
  Serial.print("Connecting to WiFi");
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("");
  Serial.println("WiFi connected!");
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());
}

void loop() {
  if (WiFi.status() == WL_CONNECTED) {
    HTTPClient http;

    // Mulai koneksi HTTP ke URL server
    http.begin(serverUrl);

    // Set Header wajib
    http.addHeader("Content-Type", "application/json");
    // Token validasi dikirim lewat header "x-api-token"
    http.addHeader("x-api-token", apiToken);

    // 1. Membaca nilai dari sensor kelembapan tanah (Analog)
    int rawMoisture = analogRead(soilPin);
    
    // Konversi nilai analog ke persen kelembapan (sesuaikan dengan kalibrasi sensor kamu)
    float humidityPercent = map(rawMoisture, 4095, 1500, 0, 100);
    humidityPercent = constrain(humidityPercent, 0, 100);

    // Dummy temperatur lingkungan (bisa diganti dengan DHT11/DHT22)
    float envTemperature = 29.7;

    // 2. Membaca data sistem internal ESP32
    // Membaca suhu chip CPU internal ESP32 (Fahrenheit ke Celcius)
    float cpuTemp = (temprature_sens_read() - 32) / 1.8; 
    int hallValue = hallRead();           // Membaca sensor medan magnet internal (Hall Sensor)
    int touchValue = touchRead(touchPin); // Membaca nilai analog Touch Pin (GPIO 4)

    // 3. Membuat JSON document sesuai skema tabel sensor di Supabase
    StaticJsonDocument<300> doc;
    doc["deviceId"] = "ESP32-Soil-Sensor-01";
    doc["moisture"] = rawMoisture;
    doc["humidity"] = humidityPercent;
    doc["temperature"] = envTemperature;
    
    doc["cpuTemperature"] = cpuTemp;
    doc["hallMagnetic"] = hallValue;
    doc["touchRaw"] = touchValue;

    // Serialisasikan JSON ke string
    String requestBody;
    serializeJson(doc, requestBody);

    Serial.println("Sending data: " + requestBody);

    // Melakukan HTTP POST request
    int httpResponseCode = http.POST(requestBody);

    if (httpResponseCode > 0) {
      String response = http.getString();
      Serial.print("HTTP Response Code: ");
      Serial.println(httpResponseCode);
      Serial.print("Response: ");
      Serial.println(response);
    } else {
      Serial.print("Error on sending POST request: ");
      Serial.println(httpResponseCode);
    }

    // Akhiri sesi HTTP
    http.end();
  } else {
    Serial.println("WiFi Disconnected. Reconnecting...");
    WiFi.begin(ssid, password);
  }

  // Kirim data setiap 10 detik (silakan sesuaikan kebutuhan)
  delay(10000);
}
```
